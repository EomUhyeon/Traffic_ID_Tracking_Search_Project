### Docker 명령어 요약

# docker images delete
sudo docker rmi -f traffic_id_tracking_search_project-frontend:latest
sudo docker rmi -f traffic_id_tracking_search_project-backend:latest

# 미사용 이미지 정리
sudo docker image prune

# frontend 이미지만 빌드 (compose 미사용)
cd frontend && sudo docker build -t traffic_id_tracking_search_project-frontend:latest . && cd ..

# frontend 서비스 빌드 및 실행
sudo docker compose up --build -d frontend

# 모든 서비스 빌드 및 실행
sudo docker compose up --build -d

# 실행중인 컨테이너 목록
sudo docker ps

# 모든 컨테이너 목록
sudo docker ps -a

# 서비스 로그 실시간 확인 (예: frontend)
sudo docker compose logs -f frontend

# 모든 서비스 중지 및 제거
sudo docker compose down
